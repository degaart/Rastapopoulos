url: ca-certificates-20160607.tar.gz
version: 20160607
extract-dir: ca-certificates-20160607
configure:
    cd "$EXTRACT_DIR"
    export PREFIX
    bash make-ca.sh
        
install:
    cd "$EXTRACT_DIR"
    export PREFIX
    export SSLDIR="$PREFIX/ssl"
    bash remove-expired-certs.sh certs
    install -dv "$SSLDIR/certs"
    cp -v certs/*.pem "$SSLDIR/certs"
    c_rehash
    install BLFS-ca-bundle*.crt "${SSLDIR}/ca-bundle.crt"
    ln -svf ../ca-bundle.crt "$SSLDIR/certs/ca-certificates.crt"

depends:
    openssl-bin
    perl > 5.8.0
