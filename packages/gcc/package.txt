url: gcc-6.3.0.tar.bz2
version: 6.3.0
extract-dir: gcc-6.3.0
configure:
    "${EXTRACT_DIR}"/configure --target=${TARGET} --prefix="${PREFIX}" --disable-nls --enable-languages=c,c++ --without-headers --with-gmp="$PREFIX" --with-mpfr="$PREFIX" --with-mpc="$PREFIX"

build:
    make -j$NJOBS all-gcc
    make -j$NJOBS all-target-libgcc

install:
    make -j$NJOBS install-gcc
    make -j$NJOBS install-target-libgcc

depends:
    binutils
    gmp >= 6.0.0
    mpfr >= 3.1.3
    mpc >= 1.0.3
