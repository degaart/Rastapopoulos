url: Python-2.7.11.tar.xz
version: 2.7.11
extract-dir: Python-2.7.11
configure:
    cd "$EXTRACT_DIR" && "$EXTRACT_DIR/configure" \
        --prefix="$PREFIX" \
        --enable-shared \
        --enable-unicode=ucs4 \
        --with-system-ffi
    
build:
    cd "$EXTRACT_DIR" && make -j${NJOBS}

install:
    cd "$EXTRACT_DIR" && make install

depends:
    libffi
    openssl
    sqlite

detect:
	if which python > /dev/null 2>&1; then
		python -V 2>&1|cut -d' ' -f2
	else
		false
	fi
